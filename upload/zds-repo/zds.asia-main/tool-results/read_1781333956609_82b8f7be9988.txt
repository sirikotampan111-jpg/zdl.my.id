   173→   173→function formatDate(dateStr: string | null | undefined): string {
   174→   174→  if (!dateStr) return "-";
   175→   175→  return new Date(dateStr).toLocaleDateString("id-ID", {
   176→   176→    day: "numeric",
   177→   177→    month: "short",
   178→   178→    year: "numeric",
   179→   179→  });
   180→   180→}
   181→   181→
   182→   182→function formatDateTime(dateStr: string | null | undefined): string {
   183→   183→  if (!dateStr) return "-";
   184→   184→  return new Date(dateStr).toLocaleDateString("id-ID", {
   185→   185→    day: "numeric",
   186→   186→    month: "short",
   187→   187→    year: "numeric",
   188→   188→    hour: "2-digit",
   189→   189→    minute: "2-digit",
   190→   190→  });
   191→   191→}

   276→   276→function MilestoneTimeline({ milestones }: { milestones: Milestone[] }) {
   277→   277→  const [expanded, setExpanded] = useState(false);
   278→   278→  if (!milestones || milestones.length === 0) return <p className="text-sm text-muted-foreground">Belum ada milestone</p>;
   279→   279→
   280→   280→  const visible = expanded ? milestones : milestones.slice(0, 3);
   281→   281→  return (
   282→   282→    <div className="space-y-2">
   283→   283→      {visible.map((m, idx) => (
   284→   284→        <div key={m.id} className="flex items-start gap-2">
   285→   285→          <div className="mt-1">
   286→   286→            {m.status === "completed" ? (
   287→   287→              <CheckCircle2 className="w-4 h-4 text-emerald-500" />
   288→   288→            ) : m.status === "in_progress" ? (
   289→   289→              <Loader2 className="w-4 h-4 text-blue-500 animate-spin" />
   290→   290→            ) : (
   291→   291→              <Clock className="w-4 h-4 text-muted-foreground" />
   292→   292→            )}
   293→   293→          </div>
   294→   294→          <div className="flex-1 min-w-0">
   295→   295→            <p className={`text-sm font-medium ${m.status === "completed" ? "line-through text-muted-foreground" : ""}`}>
   296→   296→              {m.title}
   297→   297→            </p>
   298→   298→            {m.notes && <p className="text-xs text-muted-foreground truncate">{m.notes}</p>}
   299→   299→            {m.completedAt && (
   300→   300→              <p className="text-xs text-muted-foreground">{formatDate(m.completedAt)}</p>
   301→   301→            )}
   302→   302→          </div>
   303→   303→        </div>
   304→   304→      ))}
   305→   305→      {milestones.length > 3 && (
   306→   306→        <button
   307→   307→          onClick={() => setExpanded(!expanded)}
   308→   308→          className="text-xs text-gold hover:underline flex items-center gap-1"
   309→   309→        >
   310→   310→          {expanded ? <ChevronUp className="w-3 h-3" /> : <ChevronDown className="w-3 h-3" />}
   311→   311→          {expanded ? "Tutup" : `Lihat ${milestones.length - 3} lagi`}
   312→   312→        </button>
   313→   313→      )}
   314→   314→    </div>
   315→   315→  );
   316→   316→}
